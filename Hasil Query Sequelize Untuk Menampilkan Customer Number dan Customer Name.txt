SELECT `orders->customers`.`customerNumber` AS `customerNumber`, `orders->customers`.`customerName` AS `customerName` FROM `orderdetails` AS `OrderDetails` INNER JOIN `orders` AS `orders` ON `OrderDetails`.`orderNumber` = `orders`.`orderNumber` INNER JOIN `customers` AS `orders->customers` ON `orders`.`customerNumber` = `orders->customers`.`customerNumber` INNER JOIN `products` AS `products` ON `OrderDetails`.`productCode` = `products`.`productCode` GROUP BY `orders->customers`.`customerNumber`, `orders->customers`.`customerName`;
